Intro

var obj1 = {
	test:function(){
		deep.log("this.shared : ", this.myShared);
	},
	myShared:deep.Shared([1,2,3])
};
var obj2 = {
	myShared:[4,5]
};

deep.up(obj1, obj2);
obj1.test();				// [1, 2, 3, 4, 5]
obj1.myShared.push(6,7);
obj2.test();				// [1, 2, 3, 4, 5, 6, 7]
